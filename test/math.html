# 快排

// 最简单的快排
function quick(arr) {
  let left = []
  let right = []
  let midIndex = Math.floor(arr.length / 2)
  let midValue = arr.splice(midIndex, 1)[0]
  for (let i in arr) {
    arr[i] < midValue ? left.push(arr[i]) : right.push(arr[i])
  }
  return quick(left).concat(midValue, quick(right))
}

//优化：原地快排 设定一个基准值
function quick(arr, low = 0, high = arr.length - 1) {
  let left = low
  let right = high
  //定义基准值
  let tag = arr[left]
  if (left >= right) return
  while (left < right) {
    if (left < right && tag < arr[right]) {
      right--
    }
    arr[left] = arr[right]
    if (left < right && tag > arr[left]) {
      left++
    }
    arr[right] = arr[left]
  }
  arr[left] = tag
  quick(arr, low, left - 1)
  quick(arr, left + 1, high)
  return arr
}

function swap(arr, indexA, indexB) {
  ;[arr[indexA], arr[indexB]] = [arr[indexB], arr[indexA]]
}
// 暴力递归
function bubble(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    for (let j = 0; j < i; j++) {
      if (arr[j] > arr[j + 1]) {
        swap[(arr, j, j + 1)]
      }
    }
  }
}

# 选择

function select(arr) {
  for (let i = 0; i < arr.length; i++) {
    let k = i
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[k]) k = i
    }
    if (k !== i) {
      swap(arr, i, k)
    }
  }
  return arr
}

# 插入

function insert(arr) {
  for (let i = 0; i < arr.length; i++) {
    let temp = arr[i]
    for (let j = i - 1; j >= 0 && arr[j] > temp; j--) {
      arr[j + 1] = arr[j]
    }
    arr[j + 1] = temp
  }
  return arr
}
