   _resolve(val){

         let cb;
         while((cb=this.fullfilledQueues.shift())){
           cb(val)
         }
       }
       const onRejected = (err)=>{
          let cb;
         while((cb=this.rejectedQueues.shift())){
           cb(err)
         }

         })

# 2. 手写 ajax

function ajax(options) {
  // 默认配置
  let defaults = {
    type: 'get',
    url: '',
    header: { 'Content-Type': 'application/x-www-form-urlencoded' },
    data: {},
    success: function () {},
    failed: function () {},
  }
  //覆盖默认配置
  Object.assign(defaults, options)
  let xhr = new XMLHttpRequest()
  //拼接参数
  let params = ''
  for (let key in defaults.data) {
    params += key + '=' + defaults.data[key] + '&'
  }
  //去除最后一个&
  params = params.substr(0, params.length - 1)
  //判断请求类型
  if (defaults.type === 'get') {
    defaults.url += '?' + params
  }
  xhr.open(defaults.type, defaults.url)
  if (defaults.type === 'post') {
    xhr.setRequestHeader('Content-Type', defaults.header['Content-Type'])
    if (defaults.header['Content-Type'] === 'application/json') {
      xhr.send(JSON.stringify(defaults.data))
    } else {
      xhr.send(params)
    }
  } else {
    xhr.send()
  }
  xhr.onload = function () {
    let ContentType = xhr.getResponseHeader('Content-Type')
    let responseText = xhr.responseText
    if (ContentType === 'application/json') {
      responseText = JSON.parse(responseText)
    }
    if (xhr.status === 200) {
      defaults.success(responseText, xhr)
    } else {
      defaults.failed(responseText, xhr)
    }
  }
}

Function.prototype.myapply = function (con) {
  con = new Object(con) || window || global
  con.fun = this
  let newArray = new Array()
  for (let i = 0; i < arguments[1].length; i++) {
    newArray.push(arguments[1][i])
  }
  con.fun(...newArray)
  delete con.fun
}

Function.prototype.mybind = function (con) {
  con = new Object(con) || window || global
  con.fun = this
  let newArray = new Array()
  for (let i = 1; i < arguments.length; i++) {
    newArray.push(arguments[i])
  }
  // bind返回的是一个全新的函数
  return () => {
    con.fun(...newArray)
  }
}

# 4.vue2 和 vue3 的数据劫持模拟

//vue2:
class Observer {
  constructor(obj) {
    for (let key in obj) {
      if (typeof obj[key] === 'object') {
        obj[key] = new Observer(obj[key])
      }
      Object.defineProperty(this, key, {
        // 可删除
        configurable: true,
        // 可枚举
        enumerable: true,
        get() {
          return obj[key]
        },
        set(newVal) {
          if (obj[key] === newVal) return
          obj[key] = newVal
        },
      })
    }
  }
}

# 5.手写节流防抖

  1. 节流是在频繁触发时，规定一段时间（delay）内执行一次，适用于用户输入用户名和密码时的检验格式（比如 delay 为 2 秒，则在用户不断输入的过程中，每隔 2 秒执行一次检验）
// 时间戳的写法
 function throttle(fn,delay){
   var now = Date.now()
   return function(){
     var context = this
     var next = Date.now()
     if(next-now>=delay){
       fn.apply(context)
       now = Date.now()
     }
   }
 }
 //定时器的写法
 function throttle(fn,delay){
   var timer = null
   return function(){
     var context = this
     if(!timer){
       timer = setTimeout(()=>{
         fn.apply(context)
         timer = null
       },delay)
     }
   }
 }

 //优化的写法
 function throttle(fn,delay){
   var timer = null
   var startTime = Date.now()
   return function(){
     var curTime =  Date.now()
     var remaining = delay - (curTime - startTime)
     var context = this
     clearTimeout(timer)
     if(remaining<=0)}
      //第一次触发
      fn.apply(context)
    }else{
      timer = setTimeout(fn,remaning)
    }
 }
  1. 防抖是在频繁触发时，只有在一段时间内不触发（大于 delay），才会执行，适用于滚动事件的监听（比如 delay 是 2 秒，在不断的滚动过程中，只有上下两次滚动时间超过 2 秒才会执行一次）
function debouce(fn, delay) {
  var timer = null
  return function () {
    if (timer !== null) clearTimeout(timer)
    timer = setTimeout(fn, delay)
  }
}
